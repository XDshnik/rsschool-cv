RecipeQuery("SELECT * FROM Cameras", total_row)
  if total_row == 1 then
    free_num = 1
  else
    for i = 1 to 128 step 1
      for j = 0 to total_row step 1
        RecipeGetData(num, "Cameras.number",j)
        if num == i then
          break
      end if
      next j
      if j >= total_row then
        free_num = i
        break
      end if
    next i
  end if
